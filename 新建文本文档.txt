import React, { useState, useEffect, useRef } from 'react';
import { ArrowRight, Terminal, User, Mail, Zap, Code, Play, Image as ImageIcon, Briefcase, GraduationCap, MonitorPlay, Layers } from 'lucide-react';

const App = () => {
  const [currentPage, setCurrentPage] = useState('home');
  const [isTransitioning, setIsTransitioning] = useState(false);
  const [transitionStep, setTransitionStep] = useState(0); 
  
  const cursorRef = useRef(null);
  const cursorDotRef = useRef(null);

  // 自定义光标逻辑
  useEffect(() => {
    // 动态添加 meta 标签，防止浏览器缓存
    const metaCache = document.createElement('meta');
    metaCache.httpEquiv = 'Cache-Control';
    metaCache.content = 'no-cache, no-store, must-revalidate';
    document.head.appendChild(metaCache);

    const metaPragma = document.createElement('meta');
    metaPragma.httpEquiv = 'Pragma';
    metaPragma.content = 'no-cache';
    document.head.appendChild(metaPragma);

    const metaExpires = document.createElement('meta');
    metaExpires.httpEquiv = 'Expires';
    metaExpires.content = '0';
    document.head.appendChild(metaExpires);

    const handleMouseMove = (e) => {
      if (cursorRef.current && cursorDotRef.current) {
        cursorRef.current.style.transform = `translate3d(${e.clientX - 20}px, ${e.clientY - 20}px, 0)`;
        cursorDotRef.current.style.transform = `translate3d(${e.clientX - 4}px, ${e.clientY - 4}px, 0)`;
      }
    };
    const handleMouseDown = () => { if (cursorRef.current) cursorRef.current.style.transform += ' scale(0.5)'; };
    const handleMouseUp = () => { if (cursorRef.current) cursorRef.current.style.transform = cursorRef.current.style.transform.replace(' scale(0.5)', ''); };

    window.addEventListener('mousemove', handleMouseMove);
    window.addEventListener('mousedown', handleMouseDown);
    window.addEventListener('mouseup', handleMouseUp);

    return () => {
      window.removeEventListener('mousemove', handleMouseMove);
      window.removeEventListener('mousedown', handleMouseDown);
      window.removeEventListener('mouseup', handleMouseUp);
    };
  }, []);

  const navigateTo = (page) => {
    if (page === currentPage || isTransitioning) return;
    setIsTransitioning(true);
    setTransitionStep(1); 
    setTimeout(() => {
      setCurrentPage(page); 
      setTransitionStep(2); 
      setTimeout(() => {
        setIsTransitioning(false);
        setTransitionStep(0); 
      }, 800);
    }, 800);
  };

  return (
    <div className="min-h-screen bg-[#050505] text-white font-sans overflow-x-hidden selection:bg-[#ff0055] selection:text-white">
      <style>{`
        @import url('https://fonts.googleapis.com/css2?family=Space+Grotesk:wght@400;700;900&display=swap');
        body { font-family: 'Space Grotesk', system-ui, sans-serif; cursor: none; }
        
        .glitch-text { position: relative; }
        .glitch-text::before, .glitch-text::after {
          content: attr(data-text); position: absolute; top: 0; left: 0; width: 100%; height: 100%; opacity: 0.8;
        }
        .glitch-text::before {
          color: #00ff99; z-index: -1; transform: translate(-4px, 2px); animation: glitch-anim-1 2s infinite linear alternate-reverse;
        }
        .glitch-text::after {
          color: #ff0055; z-index: -2; transform: translate(4px, -2px); animation: glitch-anim-2 3s infinite linear alternate-reverse;
        }
        @keyframes glitch-anim-1 {
          0% { clip-path: inset(20% 0 80% 0); transform: translate(-4px, 2px); }
          20% { clip-path: inset(60% 0 10% 0); transform: translate(4px, -2px); }
          40% { clip-path: inset(40% 0 50% 0); transform: translate(-4px, 2px); }
          60% { clip-path: inset(80% 0 5% 0); transform: translate(4px, -2px); }
          80% { clip-path: inset(10% 0 70% 0); transform: translate(-4px, 2px); }
          100% { clip-path: inset(30% 0 50% 0); transform: translate(4px, -2px); }
        }
        @keyframes glitch-anim-2 {
          0% { clip-path: inset(10% 0 60% 0); transform: translate(4px, -2px); }
          20% { clip-path: inset(30% 0 20% 0); transform: translate(-4px, 2px); }
          40% { clip-path: inset(70% 0 10% 0); transform: translate(4px, -2px); }
          60% { clip-path: inset(20% 0 50% 0); transform: translate(-4px, 2px); }
          80% { clip-path: inset(50% 0 30% 0); transform: translate(4px, -2px); }
          100% { clip-path: inset(5% 0 80% 0); transform: translate(-4px, 2px); }
        }
        .page-enter { animation: slideUpFade 0.8s cubic-bezier(0.16, 1, 0.3, 1) forwards; opacity: 0; transform: translateY(40px); }
        @keyframes slideUpFade { to { opacity: 1; transform: translateY(0); } }
        .hover-shake:hover { animation: shake 0.5s cubic-bezier(.36,.07,.19,.97) both; }
        @keyframes shake {
          10%, 90% { transform: translate3d(-1px, 0, 0); }
          20%, 80% { transform: translate3d(2px, 0, 0); }
          30%, 50%, 70% { transform: translate3d(-4px, 0, 0); }
          40%, 60% { transform: translate3d(4px, 0, 0); }
        }
        .hide-scrollbar::-webkit-scrollbar { display: none; }
      `}</style>

      {/* 光标 */}
      <div ref={cursorRef} className="fixed top-0 left-0 w-10 h-10 border-2 border-[#00ff99] rounded-full pointer-events-none z-[100] transition-transform duration-100 ease-out mix-blend-difference" />
      <div ref={cursorDotRef} className="fixed top-0 left-0 w-2 h-2 bg-[#ff0055] rounded-full pointer-events-none z-[100]" />

      {/* 转场遮罩 */}
      <div className="fixed inset-0 z-50 pointer-events-none flex" aria-hidden="true">
        {['#ff0055', '#00ff99', '#ffe600', '#00ccff'].map((color, index) => (
          <div key={index} className="flex-1 h-full transition-transform duration-700 ease-[cubic-bezier(0.85,0,0.15,1)] origin-bottom"
            style={{ backgroundColor: color, transform: transitionStep === 0 ? 'scaleY(0)' : transitionStep === 1 ? 'scaleY(1)' : 'scaleY(0)', transformOrigin: transitionStep === 1 ? 'bottom' : 'top', transitionDelay: `${index * 0.1}s` }} />
        ))}
      </div>

      <nav className="fixed w-full top-0 p-6 flex justify-between items-center z-40 mix-blend-difference">
        <div onClick={() => navigateTo('home')} className="text-2xl font-black cursor-pointer tracking-tighter hover:text-[#00ff99] transition-colors">
          YUNTING.<span className="text-[#ff0055]">ZHU</span>
        </div>
        <div className="flex gap-8 text-lg font-bold">
          {['home', 'works', 'resume'].map((page) => (
            <button key={page} onClick={() => navigateTo(page)} className={`uppercase tracking-widest relative group overflow-hidden ${currentPage === page ? 'text-[#ffe600]' : 'text-white'}`}>
              <span className="relative z-10 group-hover:text-[#050505] transition-colors duration-300">{page}</span>
              <div className="absolute inset-0 bg-[#00ff99] translate-y-full group-hover:translate-y-0 transition-transform duration-300 ease-in-out z-0" />
            </button>
          ))}
        </div>
      </nav>

      <main className="relative min-h-screen pt-32 pb-24 px-6 md:px-20 z-10">
        {currentPage === 'home' && <HomeView navigateTo={navigateTo} />}
        {currentPage === 'works' && <WorksView />}
        {currentPage === 'resume' && <ResumeView />}
      </main>

      {/* 极简风底部栏：小红书与B站入口 */}
      <div className="fixed bottom-0 w-full bg-[#050505] text-white py-4 px-6 md:px-20 z-30 flex flex-col md:flex-row justify-between items-center border-t-2 border-[#1a1a1a]">
        <div className="font-bold text-gray-600 text-sm md:text-base tracking-widest mb-2 md:mb-0 uppercase flex items-center gap-2">
          <Zap size={16} className="text-[#ffe600]"/> Signal_Link :: Connect
        </div>
        <div className="flex gap-8 items-center">
          <a href="XXX" target="_blank" rel="noreferrer" className="flex items-center gap-2 hover:text-[#ff0055] font-black transition-colors uppercase tracking-widest text-sm md:text-base group">
            小红书 (RED) <ArrowRight size={16} className="group-hover:translate-x-1 transition-transform" />
          </a>
          <span className="text-gray-800">|</span>
          <a href="XXX" target="_blank" rel="noreferrer" className="flex items-center gap-2 hover:text-[#00ccff] font-black transition-colors uppercase tracking-widest text-sm md:text-base group">
            B站 (Bilibili) <ArrowRight size={16} className="group-hover:translate-x-1 transition-transform" />
          </a>
        </div>
      </div>
    </div>
  );
};
